CREATE DATABASE CAALMA;

DOPO AVER CREATO IL DATABASE COPIARE TUTTO E INCOLLARE

CREATE TABLE Utente(
	Nome varchar(71),
	Cognome varchar(71),
	Login varchar(71) Not Null,
	Password varchar(71) Not Null,
	
	Constraint Utentepk 
	PRIMARY KEY (Login)

);

CREATE TABLE Ingrediente(
	Nome varchar(71),
	Quantita Integer,
	
	Constraint Ingredientepk 
	PRIMARY KEY (Nome)
);

CREATE TABLE Cocktail(
	Nome varchar(71),
	Prezzo float,
	Disponibilita Boolean,
	
	Constraint Cocktailpk 
	PRIMARY KEY (Nome)
);

CREATE TABLE Frullato(
	Nome varchar(71),
	Prezzo float,
	Disponibilita Boolean,
	
	Constraint Frullatopk 
	PRIMARY KEY (Nome)
);

CREATE TABLE ComponentiC(
	Ingrediente varchar(71),
	Cocktail varchar(71),
	
	constraint ComponentiCfk1
		foreign key (Ingrediente) references Ingrediente(Nome)
			ON DELETE CASCADE
			ON UPDATE CASCADE,
	
	constraint ComponentiCfk2
		foreign key (Cocktail) references Cocktail(Nome)
			ON DELETE CASCADE
			ON UPDATE CASCADE
	
);

CREATE TABLE ComponentiF(
	Ingrediente varchar(71),
	Frullato varchar(71),
	
	constraint ComponentiCfk1
		foreign key (Ingrediente) references Ingrediente(Nome)
			ON DELETE CASCADE
			ON UPDATE CASCADE,
	
	constraint ComponentiCfk2
		foreign key (Frullato) references Frullato(Nome)
			ON DELETE CASCADE
			ON UPDATE CASCADE
	
);

CREATE TABLE Scontrino(
	Prezzo float,
	Id Integer NOT NULL,
	Login varchar(71),
	
	Constraint Scontrinopk 
	PRIMARY KEY (id),
	
	constraint Scontrinofk1
		foreign key (Login) references Utente(Login)
			ON DELETE CASCADE
			ON UPDATE CASCADE
);

CREATE TABLE ContenutoC(
	Quantita Integer,
	Cocktail varchar(71),
	Scontrino Integer,
	
	constraint ContenutoCfk1
		foreign key (Cocktail) references Cocktail(Nome)
			ON DELETE CASCADE
			ON UPDATE CASCADE,
	
	constraint ContenutoCfk2
		foreign key (Scontrino) references Scontrino(Id)
			ON DELETE CASCADE
			ON UPDATE CASCADE
	
);

CREATE TABLE ContenutoF(
	Quantita Integer,
	Frullato varchar(71),
	Scontrino Integer,
	
	constraint ContenutoFfk1
		foreign key (Frullato) references Frullato(Nome)
			ON DELETE CASCADE
			ON UPDATE CASCADE,
	
	constraint ContenutoFfk2
		foreign key (Scontrino) references Scontrino(Id)
			ON DELETE CASCADE
			ON UPDATE CASCADE
);
